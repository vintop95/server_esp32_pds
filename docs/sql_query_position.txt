"CREATE TABLE packet("
 "id integer primary key,"
 "sender_mac text,"
 "timestamp integer,"
 "rssi integer,"
 "hashed_pkt text,"
 "ssid text,"
 "espName text)"

SELECT 		sender_mac, espName, AVG(rssi)
FROM 			packet P
WHERE			timestamp > LAST_TS
	AND			hashed_pkt IN (
			SELECT 		hashed_pkt
			FROM 			packet P1
			WHERE 		timestamp > LAST_TS
			GROUP BY	hashed_pkt
			HAVING COUNT(*) = N)
GROUP BY	sender_mac, espName
